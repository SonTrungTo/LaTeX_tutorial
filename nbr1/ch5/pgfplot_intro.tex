\begin{tikzpicture}
  \begin{axis}[
    axis lines= center,
    xlabel= $x$,
    ylabel= {$f(x)$},
    legend style={at={(axis cs:20,50)},anchor=north east},
    ]
    %We now add a red parabola
    \addplot[
    domain=-10:10,
    samples=100,
    color=red,
    ]
    {x^2-2*x-1};
    \addlegendentry{$x^2-2x-1$}
    %Now we add another parabola to the same axis
    \addplot[
    domain=-10:10,
    samples=100,
    color=blue
    ]
    {x^2+2*x+1};
    \addlegendentry{$x^2+2x+1$}
  \end{axis}
\end{tikzpicture}
